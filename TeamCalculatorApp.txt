using System;
using TeamCalculatorApp.Operations;

namespace TeamCalculatorApp
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("=== Team Calculator App ===");
            Console.WriteLine("Select Operation: +  -  *  /");
            char operation = Convert.ToChar(Console.ReadLine());

            Console.Write("Enter first number: ");
            double num1 = Convert.ToDouble(Console.ReadLine());

            Console.Write("Enter second number: ");
            double num2 = Convert.ToDouble(Console.ReadLine());

            double result = 0;

            switch (operation)
            {
                case '+':
                    result = Addition.Add(num1, num2);
                    break;
                case '-':
                    result = Subtraction.Subtract(num1, num2);
                    break;
                case '*':
                    result = Multiplication.Multiply(num1, num2);
                    break;
                case '/':
                    result = Division.Divide(num1, num2);
                    break;
                default:
                    Console.WriteLine("Invalid operation!");
                    break;
            }

            Console.WriteLine($"\nResult: {num1} {operation} {num2} = {result}");
            Console.WriteLine("\n--- Operation completed successfully by the Team ---");
        }
    }
}
